# Extend with Condition

## Definition

```js
{
  _style: {
    entity: 'shape=note;size=15;spacingLeft=5;align=left;whiteSpace=wrap;',
  },
  _width: 250,
  _height: 80,
}
```

## Usage

```js
import { ExtendWithCondition } from '@dinghy/standard-components-diagrams/sysmlUsecases'

<ExtendWithCondition/>
```
